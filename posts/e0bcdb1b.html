P6033 合并果子 加强版

2020-07-28- 2020-09-19
OI

合并果子的加强版，扩大了数据范围

  • Subtask 1（10 points）：1≤n≤81 \leq n \leq 81≤n≤8。
  • Subtask 2（20 points）：1≤n≤1031 \leq n \leq 10^31≤n≤103
  • Subtask 3（30 points）：1≤n≤1051 \leq n \leq 10^51≤n≤105
  • Subtask 4（40 points）：1≤n≤1071 \leq n \leq 10^71≤n≤107

对于全部的测试点，保证1≤ai≤105。1 \leq a_i \leq 10^5。1≤ai​≤105。

由于n太大，复杂度必须为O(N)

发现a[i]不大，可以使用桶排

不能使用优先队列，改用数组模拟

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll inf=10000000000000000;//小心inf 
#define maxn 10000009
#define f(x) x=(x==0?inf:x) 
ll n;
int t[100009];
ll st1[maxn];
ll st2[maxn];
int h1=1,t1=0;
int h2=1,t2=0;
int read()//必须快读 
{
	int x=0;
	char c=getchar();
	while(c<'0'||c>'9')
	{
		c=getchar();
	}
	while(c>='0'&&c<='9')
	{
		x=x*10+c-'0';
		c=getchar();
	}
	return x;
}
signed main()
{
	//freopen("h.in","r",stdin);
	n=read();
	ll s=0;
	for(int i=1;i<=n;i++)//通排 
	{
		t[read()]++;
	}
	for(int i=1;i<=100000;i++)
	{
		while(t[i]--)
		{
			st1[++t1]=i;
		}
	}
	for(int i=1;i<n;i++)
	{
		ll w=st1[h1];
		ll x=st1[h1+1];
		ll y=st2[h2];
		ll z=st2[h2+1];
		f(w);f(x);f(y);f(z);
		ll ans=0;
		if(w<y)//第一个用w 
		{
			ans+=w;
			h1++;
			if(x<y)
			{
				h1++;
				ans+=x;
			}else
			{
				h2++;
				ans+=y;
			}
		}else
		{
			ans+=y;
			h2++;
			if(w<z)
			{
				h1++;
				ans+=w;
			}else
			{
				h2++;
				ans+=z;
			}
		}
		s+=ans;
		st2[++t2]=ans;
	}
	printf("%lld\n",s);
	return 0;
}
